1. Bookstore BST (insert, search, delete, update)

/* Experiment 1: Online Bookstore using BST Sample Input/Output: Add: "C++ Primer", "Lippman", 45.5 Add: "Clean Code", "Martin", 50.0 Search: "Clean Code" -> Found: Clean Code | Martin | 50 Update: "C++ Primer" price to 49.99 Display (inorder): C++ Primer | Lippman | 49.99 Clean Code | Martin | 50 */ #include <iostream> #include <string> using namespace std; struct Book { string title, author; float price; Book *left, *right; Book(string t, string a, float p) : title(t), author(a), price(p), left(NULL), right(NULL) {} }; class BST { Book *root; Book* insert(Book* node, string t, string a, float p) { if (!node) return new Book(t, a, p); if (t < node->title) node->left = insert(node->left, t, a, p); else if (t > node->title) node->right = insert(node->right, t, a, p); return node; } Book* search(Book* node, string t) { if (!node || node->title == t) return node; if (t < node->title) return search(node->left, t); return search(node->right, t); } Book* findMin(Book* node) { while (node && node->left) node = node->left; return node; } Book* remove(Book* node, string t) { if (!node) return NULL; if (t < node->title) node->left = remove(node->left, t); else if (t > node->title) node->right = remove(node->right, t); else { if (!node->left) { Book* temp = node->right; delete node; return temp; } if (!node->right) { Book* temp = node->left; delete node; return temp; } Book* temp = findMin(node->right); node->title = temp->title; node->author = temp->author; node->price = temp->price; node->right = remove(node->right, temp->title); } return node; } void inorder(Book* node) { if (!node) return; inorder(node->left); cout << node->title << " | " << node->author << " | $" << node->price << endl; inorder(node->right); } public: BST() : root(NULL) {} void add(string t, string a, float p) { root = insert(root, t, a, p); } void searchBook(string t) { Book* res = search(root, t); if (res) cout << "Found: " << res->title << " | " << res->author << " | $" << res->price << endl; else cout << "Not found\n"; } void updatePrice(string t, float newPrice) { Book* res = search(root, t); if (res) { res->price = newPrice; cout << "Updated\n"; } else cout << "Not found\n"; } void deleteBook(string t) { root = remove(root, t); cout << "Deleted (if existed)\n"; } void display() { inorder(root); } }; int main() { BST store; store.add("C++ Primer", "Lippman", 45.5); store.add("Clean Code", "Martin", 50.0); store.searchBook("Clean Code"); store.updatePrice("C++ Primer", 49.99); store.display(); store.deleteBook("Clean Code"); return 0; }

/* Experiment 2: Threaded Binary Tree Sample Input/Output: Insert: 20, 10, 30, 5, 15 Inorder: 5 10 15 20 30 Search 15: Found */ #include<iostream> using namespace std; struct node { int data; node *left, *right; int lthread, rthread; }; node *root = NULL; void insert(int key) { node *ptr = root, *parent = NULL; while (ptr) { if (key == ptr->data) { cout << "Duplicate\n"; return; } parent = ptr; if (key < ptr->data) { if (ptr->lthread == 0) ptr = ptr->left; else break; } else { if (ptr->rthread == 0) ptr = ptr->right; else break; } } node *nn = new node{key, NULL, NULL, 1, 1}; if (!parent) { root = nn; return; } if (key < parent->data) { nn->left = parent->left; nn->right = parent; parent->lthread = 0; parent->left = nn; } else { nn->right = parent->right; nn->left = parent; parent->rthread = 0; parent->right = nn; } } void inorder() { if (!root) return; node *ptr = root; while (ptr->lthread == 0) ptr = ptr->left; while (ptr) { cout << ptr->data << " "; if (ptr->rthread == 1) ptr = ptr->right; else { ptr = ptr->right; while (ptr && ptr->lthread == 0) ptr = ptr->left; } } } void search(int key) { node *ptr = root; while (ptr) { if (key == ptr->data) { cout << "Found\n"; return; } if (key < ptr->data) { if (ptr->lthread == 0) ptr = ptr->left; else break; } else { if (ptr->rthread == 0) ptr = ptr->right; else break; } } cout << "Not found\n"; } int main() { int arr[] = {20,10,30,5,15}; for(int x: arr) insert(x); cout << "Inorder: "; inorder(); cout << endl; search(15); search(100); return 0; }

3. AVL Tree (insert, search, balance)

📋 Sample run in comments
/* Experiment 3: AVL Tree (Height Balanced) Sample Input/Output: Insert 10,20,30 → triggers rotations Inorder: 10 20 30 Search 20: Found */ #include<iostream> using namespace std; struct Node { int data, height; Node *left, *right; Node(int val) : data(val), height(1), left(NULL), right(NULL) {} }; int height(Node *n) { return n ? n->height : 0; } int getBal(Node *n) { return n ? height(n->left) - height(n->right) : 0; } Node* rightRotate(Node *y) { Node *x = y->left, *T2 = x->right; x->right = y; y->left = T2; y->height = max(height(y->left), height(y->right)) + 1; x->height = max(height(x->left), height(x->right)) + 1; return x; } Node* leftRotate(Node *x) { Node *y = x->right, *T2 = y->left; y->left = x; x->right = T2; x->height = max(height(x->left), height(x->right)) + 1; y->height = max(height(y->left), height(y->right)) + 1; return y; } Node* insert(Node* root, int val) { if (!root) return new Node(val); if (val < root->data) root->left = insert(root->left, val); else if (val > root->data) root->right = insert(root->right, val); else return root; root->height = 1 + max(height(root->left), height(root->right)); int bal = getBal(root); if (bal > 1 && val < root->left->data) return rightRotate(root); if (bal < -1 && val > root->right->data) return leftRotate(root); if (bal > 1 && val > root->left->data) { root->left = leftRotate(root->left); return rightRotate(root); } if (bal < -1 && val < root->right->data) { root->right = rightRotate(root->right); return leftRotate(root); } return root; } void inorder(Node *r) { if(r) { inorder(r->left); cout << r->data << " "; inorder(r->right); } } bool search(Node *r, int val) { if(!r) return false; if(r->data == val) return true; return val < r->data ? search(r->left, val) : search(r->right, val); } int main() { Node *root = NULL; int vals[] = {10,20,30,40,50,25}; for(int v: vals) root = insert(root, v); cout << "Inorder: "; inorder(root); cout << endl; cout << "Search 30: " << (search(root,30)?"Found":"Not found") << endl; return 0; }

/* Experiment 10: Dijkstra's Algorithm Input: cities=5, roads, source=0 dest=4 Output: Shortest distance and path. */ #include<iostream> #define INF 999 using namespace std; int graph[10][10], n; void dijkstra(int src, int dest) { int dist[10], visited[10]={0}, parent[10]; for(int i=0;i<n;i++) dist[i]=INF; dist[src]=0; parent[src]=-1; for(int c=0;c<n-1;c++) { int u=-1, minD=INF; for(int i=0;i<n;i++) if(!visited[i] && dist[i]<minD) { minD=dist[i]; u=i; } if(u==-1) break; visited[u]=1; for(int v=0;v<n;v++) if(!visited[v] && graph[u][v] && dist[u]+graph[u][v] < dist[v]) { dist[v]=dist[u]+graph[u][v]; parent[v]=u; } } cout << "Distance: " << dist[dest] << "\nPath: "; int path[10], idx=0, cur=dest; while(cur!=-1) { path[idx++]=cur; cur=parent[cur]; } for(int i=idx-1;i>=0;i--) cout << path[i] << " "; cout << endl; } int main() { int e, src, dest; cout << "Cities edges: "; cin >> n >> e; for(int i=0;i<e;i++) { int u,v,w; cin >> u >> v >> w; graph[u][v]=graph[v][u]=w; } cout << "Source destination: "; cin >> src >> dest; dijkstra(src, dest); return 0; }

/* Experiment 13: OBST (Dynamic Programming) Sample: keys=10,20,30 freq=4,2,6 -> min cost=26 */ #include<iostream> #define MAX 10 using namespace std; int keys[MAX], freq[MAX], n; int cost[MAX][MAX], root[MAX][MAX]; void compute() { for(int i=0;i<n;i++) { cost[i][i]=freq[i]; root[i][i]=i; } for(int len=2;len<=n;len++) { for(int i=0;i<=n-len;i++) { int j=i+len-1, sum=0; for(int k=i;k<=j;k++) sum+=freq[k]; cost[i][j]=99999; for(int r=i;r<=j;r++) { int c = ((r>i)?cost[i][r-1]:0) + ((r<j)?cost[r+1][j]:0) + sum; if(c < cost[i][j]) { cost[i][j]=c; root[i][j]=r; } } } } cout << "Minimum cost: " << cost[0][n-1] << endl; } int main() { cout << "Number of keys: "; cin >> n; cout << "Keys: "; for(int i=0;i<n;i++) cin >> keys[i]; cout << "Frequencies: "; for(int i=0;i<n;i++) cin >> freq[i]; compute(); return 0; }
