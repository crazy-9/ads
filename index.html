/* Experiment 1: Online Bookstore using BST Sample Input/Output: 1. Add: "C++ Primer", "Lippman", 45.5 2. Add: "Clean Code", "Martin", 50.0 3. Search "Clean Code" -> Found 4. Update "C++ Primer" price to 49.99 5. Display (inorder) */ #include <iostream> #include <cstring> using namespace std; struct Book { char title[50], author[50]; float price; Book *left, *right; }; Book* createBook(char t[], char a[], float p) { Book* b = new Book; strcpy(b->title, t); strcpy(b->author, a); b->price = p; b->left = b->right = NULL; return b; } Book* insert(Book* root, char t[], char a[], float p) { if (!root) return createBook(t, a, p); if (strcmp(t, root->title) < 0) root->left = insert(root->left, t, a, p); else if (strcmp(t, root->title) > 0) root->right = insert(root->right, t, a, p); return root; } Book* search(Book* root, char t[]) { if (!root || strcmp(root->title, t) == 0) return root; if (strcmp(t, root->title) < 0) return search(root->left, t); return search(root->right, t); } void inorder(Book* root) { if (!root) return; inorder(root->left); cout << root->title << " | " << root->author << " | $" << root->price << endl; inorder(root->right); } int main() { Book* root = NULL; int choice; char title[50], author[50]; float price; do { cout << "\n1.Add 2.Search 3.Update 4.Display 5.Exit\nChoice: "; cin >> choice; cin.ignore(); if (choice == 1) { cout << "Title: "; cin.getline(title, 50); cout << "Author: "; cin.getline(author, 50); cout << "Price: "; cin >> price; root = insert(root, title, author, price); } else if (choice == 2) { cout << "Title: "; cin.getline(title, 50); Book* res = search(root, title); if (res) cout << "Found: " << res->title << " | " << res->author << " | $" << res->price << endl; else cout << "Not found\n"; } else if (choice == 3) { cout << "Title: "; cin.getline(title, 50); Book* res = search(root, title); if (res) { cout << "New price: "; cin >> price; res->price = price; cout << "Updated\n"; } else cout << "Not found\n"; } else if (choice == 4) { inorder(root); } } while (choice != 5); return 0; }

2. Threaded Binary Tree – insert, inorder, search

/* Experiment 2: Threaded Binary Tree Sample Input: 20, 10, 30, 5, 15 Inorder: 5 10 15 20 30 Search 15: Found */ #include<iostream> using namespace std; struct node { int data; node *left, *right; int lthread, rthread; }; node *root = NULL; void insert(int key) { node *ptr = root, *parent = NULL; while (ptr) { if (key == ptr->data) { cout << "Duplicate\n"; return; } parent = ptr; if (key < ptr->data) { if (ptr->lthread == 0) ptr = ptr->left; else break; } else { if (ptr->rthread == 0) ptr = ptr->right; else break; } } node *nn = new node{key, NULL, NULL, 1, 1}; if (!parent) { root = nn; return; } if (key < parent->data) { nn->left = parent->left; nn->right = parent; parent->lthread = 0; parent->left = nn; } else { nn->right = parent->right; nn->left = parent; parent->rthread = 0; parent->right = nn; } } void inorder() { if (!root) return; node *ptr = root; while (ptr->lthread == 0) ptr = ptr->left; while (ptr) { cout << ptr->data << " "; if (ptr->rthread == 1) ptr = ptr->right; else { ptr = ptr->right; while (ptr && ptr->lthread == 0) ptr = ptr->left; } } } void search(int key) { node *ptr = root; while (ptr) { if (key == ptr->data) { cout << "Found\n"; return; } if (key < ptr->data) { if (ptr->lthread == 0) ptr = ptr->left; else break; } else { if (ptr->rthread == 0) ptr = ptr->right; else break; } } cout << "Not found\n"; } int main() { int choice, val; do { cout << "\n1.Insert 2.Inorder 3.Search 4.Exit\nChoice: "; cin >> choice; if (choice == 1) { cin >> val; insert(val); } else if (choice == 2) { inorder(); cout << endl; } else if (choice == 3) { cin >> val; search(val); } } while (choice != 4); return 0; }

3. AVL Tree – insert, search, balance

/* Experiment 3: AVL Tree (Height Balanced) Sample: insert 10,20,30 (rotations happen) Inorder: 10 20 30 Search 20: Found */ #include<iostream> using namespace std; struct Node { int data, height; Node *left, *right; Node(int val) : data(val), height(1), left(NULL), right(NULL) {} }; int height(Node *n) { return n ? n->height : 0; } int getBal(Node *n) { return n ? height(n->left) - height(n->right) : 0; } Node* rightRotate(Node *y) { Node *x = y->left, *T2 = x->right; x->right = y; y->left = T2; y->height = max(height(y->left), height(y->right)) + 1; x->height = max(height(x->left), height(x->right)) + 1; return x; } Node* leftRotate(Node *x) { Node *y = x->right, *T2 = y->left; y->left = x; x->right = T2; x->height = max(height(x->left), height(x->right)) + 1; y->height = max(height(y->left), height(y->right)) + 1; return y; } Node* insert(Node* root, int val) { if (!root) return new Node(val); if (val < root->data) root->left = insert(root->left, val); else if (val > root->data) root->right = insert(root->right, val); else return root; root->height = 1 + max(height(root->left), height(root->right)); int bal = getBal(root); if (bal > 1 && val < root->left->data) return rightRotate(root); if (bal < -1 && val > root->right->data) return leftRotate(root); if (bal > 1 && val > root->left->data) { root->left = leftRotate(root->left); return rightRotate(root); } if (bal < -1 && val < root->right->data) { root->right = rightRotate(root->right); return leftRotate(root); } return root; } void inorder(Node *r) { if(r) { inorder(r->left); cout << r->data << " "; inorder(r->right); } } bool search(Node *r, int val) { if(!r) return false; if(r->data == val) return true; return val < r->data ? search(r->left, val) : search(r->right, val); } int main() { Node *root = NULL; int choice, val; do { cout << "\n1.Insert 2.Inorder 3.Search 4.Exit\nChoice: "; cin >> choice; if (choice == 1) { cin >> val; root = insert(root, val); } else if (choice == 2) { inorder(root); cout << endl; } else if (choice == 3) { cin >> val; cout << (search(root,val)?"Found\n":"Not found\n"); } } while (choice != 4); return 0; }

/* Experiment 4: BFS on Graph Input: vertices=5, edges: 0-1,0-2,1-3,1-4 Output: BFS from 0: 0 1 2 3 4 */ #include<iostream> using namespace std; #define MAX 5 int adj[MAX][MAX], visited[MAX], n, e; int queue[MAX], front=0, rear=-1; void enq(int v) { queue[++rear]=v; } int deq() { return queue[front++]; } bool empty() { return front>rear; } void BFS(int start) { visited[start]=1; cout << start << " "; enq(start); while(!empty()) { int u = deq(); for(int i=0; i<n; i++) if(adj[u][i] && !visited[i]) { visited[i]=1; cout << i << " "; enq(i); } } } int main() { cout << "Vertices: "; cin >> n; cout << "Edges: "; cin >> e; cout << "Enter edges (u v):\n"; for(int i=0;i<e;i++) { int u,v; cin >> u >> v; adj[u][v]=adj[v][u]=1; } cout << "BFS: "; BFS(0); return 0; }

/* Experiment 9: Kruskal's MST using DSU */ #include<iostream> #include<algorithm> using namespace std; struct Edge { int u, v, w; }; bool cmp(Edge a, Edge b) { return a.w < b.w; } int parent[10]; int find(int i) { while(parent[i]>=0) i=parent[i]; return i; } void Union(int i, int j) { int ri=find(i), rj=find(j); if(ri!=rj) parent[ri]=rj; } int main() { int n, e; cout << "Vertices edges: "; cin >> n >> e; Edge edges[20]; for(int i=0;i<e;i++) cin >> edges[i].u >> edges[i].v >> edges[i].w; sort(edges, edges+e, cmp); for(int i=0;i<n;i++) parent[i]=-1; int mstCost=0; cout << "MST edges:\n"; for(int i=0;i<e;i++) { int u=edges[i].u, v=edges[i].v; if(find(u)!=find(v)) { cout << u << "-" << v << " : " << edges[i].w << endl; mstCost+=edges[i].w; Union(u,v); } } cout << "Total cost: " << mstCost << endl; return 0; }

/* Experiment 10: Dijkstra's Algorithm */ #include<iostream> #define INF 999 using namespace std; int graph[10][10], n; void dijkstra(int src, int dest) { int dist[10], visited[10]={0}, parent[10]; for(int i=0;i<n;i++) dist[i]=INF; dist[src]=0; parent[src]=-1; for(int c=0;c<n-1;c++) { int u=-1, minD=INF; for(int i=0;i<n;i++) if(!visited[i] && dist[i]<minD) { minD=dist[i]; u=i; } if(u==-1) break; visited[u]=1; for(int v=0;v<n;v++) if(!visited[v] && graph[u][v] && dist[u]+graph[u][v] < dist[v]) { dist[v]=dist[u]+graph[u][v]; parent[v]=u; } } cout << "Distance: " << dist[dest] << "\nPath: "; int path[10], idx=0, cur=dest; while(cur!=-1) { path[idx++]=cur; cur=parent[cur]; } for(int i=idx-1;i>=0;i--) cout << path[i] << " "; cout << endl; } int main() { int e, src, dest; cout << "Cities edges: "; cin >> n >> e; for(int i=0;i<e;i++) { int u,v,w; cin >> u >> v >> w; graph[u][v]=graph[v][u]=w; } cout << "Source destination: "; cin >> src >> dest; dijkstra(src, dest); return 0; }

/* Experiment 13: OBST (Dynamic Programming) Sample: keys=10,20,30 freq=4,2,6 -> min cost=26 */ #include<iostream> #define MAX 10 using namespace std; int keys[MAX], freq[MAX], n; int cost[MAX][MAX], root[MAX][MAX]; void compute() { for(int i=0;i<n;i++) { cost[i][i]=freq[i]; root[i][i]=i; } for(int len=2;len<=n;len++) { for(int i=0;i<=n-len;i++) { int j=i+len-1, sum=0; for(int k=i;k<=j;k++) sum+=freq[k]; cost[i][j]=99999; for(int r=i;r<=j;r++) { int c = ((r>i)?cost[i][r-1]:0) + ((r<j)?cost[r+1][j]:0) + sum; if(c < cost[i][j]) { cost[i][j]=c; root[i][j]=r; } } } } cout << "Minimum cost: " << cost[0][n-1] << endl; } int main() { cout << "Number of keys: "; cin >> n; cout << "Keys: "; for(int i=0;i<n;i++) cin >> keys[i]; cout << "Frequencies: "; for(int i=0;i<n;i++) cin >> freq[i]; compute(); return 0; }
